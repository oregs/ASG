        socket.emit('OnlineUser', {
            'user_id' :  $("#onlineid").val()
        });

    function doOnline(socket){
        var onlineid = $("#onlineid").val();
        if(onlineid != ""){
            online(socket);
            setTimeout(
                function(){ 
                    doOnline(socket);
            }, 5000);
        }
    }
    try {
        socket.on('newMessage', function(response) {
            var onlineid = $("#onlineid").val();
            if (response.toid == onlineid ){
                var toid = $("#toid").val();
                if(toid == "" || response.fromid != toid){
                    showNotice( tmpl('msg_notice',{
                        'message':'New message from' + response.fromuser,
                        'user_id':response.fromid,
                    }),true,'topLeft');
                }
            }
        });
        socket.on('OnlineUser', function(response) {
            var toid = $("#toid").val()
            if (response.user_id == toid){
                $('#online').val('1');
                $('#online_status').html('online');
                $('#onlinedot').html('<span class="online dot"></span>');
            }
        });
        doOnline(socket);

        console.log('Error connecting to node server');
